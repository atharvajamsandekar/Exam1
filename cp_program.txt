#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

int main() {
    int src, dest;
    char buffer[100];
    int bytes;

    src = open("file1.txt", O_RDONLY);
    dest = open("file2.txt", O_WRONLY | O_CREAT, 0644);

    while ((bytes = read(src, buffer, sizeof(buffer))) > 0) {
        write(dest, buffer, bytes);
    }

    close(src);
    close(dest);

    printf("File copied successfully\n");

    return 0;
}
