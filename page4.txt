Implementation of First Come First Served Disk Scheduling Algorithm. 

code-

# FCFS Disk Scheduling

requests = [98, 183, 37, 122, 14, 124, 65, 67]
head = 53

total_movement = 0

print("Order of execution:")
print(head, end=" -> ")

for req in requests:
    print(req, end=" -> ")
    total_movement += abs(head - req)
    head = req

print("\n\nTotal Head Movement =", total_movement)

Implementation of Shortest Seek Time First Disk Scheduling Algorithm

code-

# SSTF Disk Scheduling

requests = [98, 183, 37, 122, 14, 124, 65, 67]
head = 53

total_movement = 0
sequence = []

while requests:
    # find closest request
    nearest = min(requests, key=lambda x: abs(x - head))
    
    sequence.append(nearest)
    total_movement += abs(head - nearest)
    
    head = nearest
    requests.remove(nearest)

print("Order:", sequence)
print("Total Head Movement =", total_movement)

 Implementation of SCAN (Elevator) Disk Scheduling Algorithm.

code-

# SCAN Disk Scheduling

requests = [98, 183, 37, 122, 14, 124, 65, 67]
head = 53
disk_size = 200
direction = "left"   # left means towards 0

left = []
right = []

for r in requests:
    if r < head:
        left.append(r)
    else:
        right.append(r)

left.sort(reverse=True)
right.sort()

sequence = []

if direction == "left":
    sequence = left + [0] + right
else:
    sequence = right + [disk_size - 1] + left

total = 0
current = head

for r in sequence:
    total += abs(current - r)
    current = r

print("Order:", sequence)
print("Total Head Movement =", total)
